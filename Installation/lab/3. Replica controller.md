```
apiVersion: v1
kind: ReplicationController
metadata:
  name: nginx-rc
  labels:
    env: dev
spec:
  replicas: 2
  template:
    metadata:
      name: label-pod1
      labels:
        app: nginx
        env: prod
    spec:
      containers:
        - name: boxone
          image: nginx:latest
```
